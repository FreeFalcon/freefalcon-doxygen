164  void Rotate(const vector3& axis, float angle)
165  {
166  // rotates this one around given vector. We do
167  // rotation with matrices, but these aren't defined yet!
168  float rotM[9];
169  float sa, ca;
170 
171  sa = sinf(angle);
172  ca = cosf(angle);
173 
174  // build a rotation matrix
175  rotM[0] = ca + (1 - ca) * axis.x * axis.x;
176  rotM[1] = (1 - ca) * axis.x * axis.y - sa * axis.z;
177  rotM[2] = (1 - ca) * axis.z * axis.x + sa * axis.y;
178  rotM[3] = (1 - ca) * axis.x * axis.y + sa * axis.z;
179  rotM[4] = ca + (1 - ca) * axis.y * axis.y;
180  rotM[5] = (1 - ca) * axis.y * axis.z - sa * axis.x;
181  rotM[6] = (1 - ca) * axis.z * axis.x - sa * axis.y;
182  rotM[7] = (1 - ca) * axis.y * axis.z + sa * axis.x;
183  rotM[8] = ca + (1 - ca) * axis.z * axis.z;
184 
185  // "handmade" multiplication
186  vector3 help(rotM[0] * this->x + rotM[1] * this->y + rotM[2] * this->z,
187  rotM[3] * this->x + rotM[4] * this->y + rotM[5] * this->z,
