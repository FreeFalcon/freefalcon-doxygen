142 # define ARCHIVE_TABLE_SIZE 53 /* size of hash table to start with when
143 opening an archive. since there is no
144 simple entry in a zip file containing
145 the number of entries within a zip file,
146  and since counting the entries is a
147  rather laborious process - this value
148  is used as the starting point. If the
149  number of entries exceed this value,
150  the hash table size is dynamically
151  resized. */
152 #else /* otherwise, heirarchical */
153 # define HASH_TABLE_SIZE 29 /* only needs to hold directory names */
154 
155 # define ARCHIVE_TABLE_SIZE 29 /* if you are using hierarchies, and
156 you have recursed directory pathnames
157 into the zip, you can probably use the
158 same size hash table for archive table
159 as what you defined as HASH_TABLE_SIZE */
160 #endif /* RES_USE_FLAT_MODEL */

162  /* ----------------------------------------------------------------------
163 
164  HASH_TABLE_SIZE is the default size for the main hash table. The hash
165  table will grow if you try to put too many entries into it, but this
166  is the starting point. I try to maintain about 125% hash size vs.
167  number of entries, but I only trigger a resize of the hash table when
168  you get to 80% hash size vs. number of entries (20% more entries than
169  slots).
170 
171  ---------------------------------------------------------------------- */
